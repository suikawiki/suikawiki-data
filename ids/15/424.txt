sub scale_rgb ($$$$$) {
  my ($val1 => $val2, $color1 => $color2, $value) = @_;
  my $r = [];
  $r->[$_] = $color1->[$_] + ($color2->[$_] - $color1->[$_]) * ($value - $val1)
/ ($val2 - $val1 or 1) for 0..2;
  return $r;
} # scale_rgb
