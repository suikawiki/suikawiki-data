> <OBJECT CLASSID="CLSID:CF7DAC31-D63B-11D2-837B-00A0C95AB0A4"
>  WIDTH=400 HEIGHT=300>
> <PARAM NAME="SRC" VALUE="anime.eva">
> <PARAM NAME="MENU" VALUE="0">
> <EMBED SRC="anime.eva" WIDTH=400 HEIGHT=300 
>  PLUGINSPAGE="http://www.sharp.co.jp/sc/excite/evademo/evahome.htm"
