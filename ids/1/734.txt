[1] 情報交換用の機関名の省略方法が [[JISX0801]], [[JISX0802]]
で規定されています。

[2] で、それの script を書いてみたんですが...
[[Perl]] module 化したかったんですが、現実的に曖昧すぎて機械処理には向かないと思うので、諦めました。

[3] >>2 曖昧過ぎるってのは [[JIS]] 
の規定が悪いんじゃなくて、[[自然言語]]の処理
(単語分解とか。) や多くの語彙情報が必要になるんで、ちょこちょこっと算法を
code にするだけってわけにはいかないってことです。

[4] そのとりあえず動く script:
[PRE[
use strict;
use 5.7.3;
use encoding "euc-jp", STDOUT => 'sjis';

## JIS X 0802-1989
sub abbr_0802 ($) {
  my $s = shift;
  ## 4.4 (1) (a) 国立の試験研究機関が、庁名又は院名をもつ場合には、
  ##             その上位期間である府名又は省名を省略してもよい。
  $s =~ s/^.+[府省](?=.+[庁廰院].+所)//;
  ## 4.4 (1) (b) 公共企業体, 特殊法人などの附属機関名は、
  ##             識別性に欠けない場合に限り、法人名を省略してもよい。
  ## Note: Since I can't write regexp for this rule,
  ##       only examples of JIS are abbreved:-)
  $s =~ s/宇宙開発事業団(?=種子島宇宙センター)//;
  $s =~ s/労働福祉事業団(?=東京労災病院)//;
  $s =~ s/住宅・都市整備公団(?=住宅都市試験研究所)//;
  ## 4.4 (2) 教育機関に附属する研究機関・病院などは、
  ##         “附属”という語を省略してもよい。
  $s =~ s/(大学|部)[附付]属(?!.*学校)/$1/;
  ## 4.4 (3) 公益法人であることを示す次のような語は、
  ##         識別性に欠けない場合に限り省略してもよい。
  ## (独立行政法人は JIS になし。)
  $s =~ s/(財団|社団|宗教|医療|学校|独立行政)法人//;
  ## 4.4 (4) 民間企業の機関名の一部である“株式会社”,
  ##        “合資会社”などの語は、
  ##         識別性に欠けない場合に限り省略してもよい。
  $s =~ s/(株式|有限|合[資名])会社//;
  $s;
}

## JIS X 0801-1989
sub abbr_0801 ($) {
  my $s = shift;
  $s =~ s/([工商])業大学/$1大/;
  $s =~ s/([工商法])科大学/$1大/;
  $s =~ s/(獣医|文理)科大学/$1大/;
  $s =~ s/(商船)大学/$1大/;
  $s =~ s/医科歯科大学/医科歯科大/;
  $s =~ s/工芸繊維大学/工繊大/;
  $s =~ s/電気通信大学/電通大/;
  $s =~ s/短期大学/短大/;
  $s =~ s/([小中大])学校?/$1/;	## JIS doesn't mention [小中] because of its scope
  $s =~ s/高等専門学校/高専/;
  $s =~ s/高等学校/高/;	## JIS doesn't mention because of its scope
  $s =~ s/(総合|中央)(研究所|試験[所場])/substr ($1, 0, 1).substr ($1, 2, 1)/e;
  $s =~ s/([国都道府県市区町村]|組合)立/$1/;
  $s =~ s/([工蚕農茶産窯林])業/$1/ge;
  $s =~ s/[医化科工農理]学/$1/ge;
  $s =~ s/(研究所|試験[所場]|衛生|園芸|技術|蚕糸|[水畜]産|学術|技術|研究|彙報|紀要|集報|集録|報告[書集]?)/substr $1, 0, 1/ge;
  $s =~ s/(論文[誌集])/substr ($1, 0, 1).substr ($1, 2, 1)/ge;
  $s =~ s/通商産業省|農林水産省|工業技術院|経済企画庁|科学技術庁|文部科学省/substr ($1, 0, 1).substr ($1, 2, 1).substr ($1, 4, 1)/e;
  $s =~ s/株式会社//;
  $s;
}

my %test = (
	"総理府科学技術庁航空宇宙技術研究所"	=> "科学技術庁航空宇宙技術研究所",
	"運輸省気象庁地磁気観測所"	=> "気象庁地磁気観測所",
	"通商産業省工業技術院電子技術総合研究所"	=> "工業技術院電子技術総合研究所",
	"宇宙開発事業団種子島宇宙センター"	=> "種子島宇宙センター",
	"大阪大学医学部附属病院"	=> "大阪大学医学部病院",
	"東京医科歯科大学歯学部附属病院"	=> "東京医科歯科大学歯学部病院",
	"京都大学農学部附属演習林"	=> "京都大学農学部演習林",
	"島根大学理学部附属臨海実験所"	=> "島根大学理学部臨海実験所",
	"九州大学工学部附属分析機器センター"	=> "九州大学工学部分析機器センター",
	"富山大学教育学部附属中学校"	=> "富山大学教育学部附属中学校",
	"財団法人日本規格協会"	=> "日本規格協会",
	"独立行政法人大学入試センター"	=> "大学入試センター",
	"株式会社日立製作所"	=> "日立製作所",
	"三菱電機株式会社"	=> "三菱電機",
);

for my $word (keys %test) {
  my $aword = &abbr_0802 ($word);
  print scalar ($test{$word} eq $aword ?1:0)." $word	=> $aword\n";
}
